## Marker / MarkerF

The `Marker` component renders a marker on the map at a specified location. Both `Marker` and `MarkerF` are available - they are functionally identical (MarkerF is the functional component, Marker is an alias).

## Props

<ParamField path="position" type="google.maps.LatLng | google.maps.LatLngLiteral" required>
  The marker position. Required.
</ParamField>

<ParamField path="options" type="google.maps.MarkerOptions">
  Marker options object.
</ParamField>

<ParamField path="animation" type="google.maps.Animation">
  Animation for the marker (BOUNCE or DROP).
</ParamField>

<ParamField path="clickable" type="boolean">
  Whether the marker is clickable.
</ParamField>

<ParamField path="cursor" type="string">
  Mouse cursor to show on hover.
</ParamField>

<ParamField path="draggable" type="boolean">
  Whether the marker can be dragged.
</ParamField>

<ParamField path="icon" type="string | google.maps.Icon | google.maps.Symbol">
  Icon for the marker.
</ParamField>

<ParamField path="label" type="string | google.maps.MarkerLabel">
  Label for the marker.
</ParamField>

<ParamField path="opacity" type="number">
  Opacity of the marker (0.0 to 1.0).
</ParamField>

<ParamField path="shape" type="google.maps.MarkerShape">
  Shape for the marker's clickable region.
</ParamField>

<ParamField path="title" type="string">
  Rollover text.
</ParamField>

<ParamField path="visible" type="boolean">
  Whether the marker is visible.
</ParamField>

<ParamField path="zIndex" type="number">
  Z-index for stacking order.
</ParamField>

<ParamField path="clusterer" type="Clusterer | GoogleClusterer">
  Marker clusterer instance for clustering markers.
</ParamField>

<ParamField path="noClustererRedraw" type="boolean">
  Prevents redraw when adding/removing from clusterer.
</ParamField>

<ParamField path="children" type="ReactNode">
  Child components (e.g., InfoWindow).
</ParamField>

## Event Handlers

<ParamField path="onClick" type="(e: google.maps.MapMouseEvent) => void">
  Fired when the marker is clicked.
</ParamField>

<ParamField path="onDblClick" type="(e: google.maps.MapMouseEvent) => void">
  Fired when the marker is double-clicked.
</ParamField>

<ParamField path="onDrag" type="(e: google.maps.MapMouseEvent) => void">
  Fired repeatedly while the marker is being dragged.
</ParamField>

<ParamField path="onDragEnd" type="(e: google.maps.MapMouseEvent) => void">
  Fired when the marker is finished being dragged.
</ParamField>

<ParamField path="onDragStart" type="(e: google.maps.MapMouseEvent) => void">
  Fired when the user starts dragging the marker.
</ParamField>

<ParamField path="onMouseDown" type="(e: google.maps.MapMouseEvent) => void">
  Fired when mouse button is pressed on the marker.
</ParamField>

<ParamField path="onMouseOut" type="(e: google.maps.MapMouseEvent) => void">
  Fired when the mouse leaves the marker area.
</ParamField>

<ParamField path="onMouseOver" type="(e: google.maps.MapMouseEvent) => void">
  Fired when the mouse enters the marker area.
</ParamField>

<ParamField path="onMouseUp" type="(e: google.maps.MapMouseEvent) => void">
  Fired when mouse button is released on the marker.
</ParamField>

<ParamField path="onRightClick" type="(e: google.maps.MapMouseEvent) => void">

  Called when the component unmounts.

## Example

```tsx
import { GoogleMap, Marker } from '@react-google-maps/api';

function MyMap() {
  const position: google.maps.LatLngLiteral = { lat: 40.7128, lng: -74.0060 };

  return (
    <GoogleMap
      center={position}
      zoom={12}
      mapContainerStyle={{ width: '100%', height: '400px' }}
    >
      <Marker
        position={position}
        title="New York City"
        onClick={(e) => console.log('Marker clicked', e)}
      />
    </GoogleMap>
  );
}
```
